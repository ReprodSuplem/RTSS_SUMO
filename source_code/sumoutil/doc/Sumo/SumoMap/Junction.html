def scanXml(xml)
  @id = SumoMap.getAttrVal(xml, "id") ;
  @type = SumoMap.getAttrVal(xml, "type") ;

  x = SumoMap.getAttrVal(xml, "x").to_f ;
  y = SumoMap.getAttrVal(xml, "y").to_f ;

  @pos = Geo2D::Point.new(x, y) ;

  return self ;
end

def toJson()
  json = { 'class' => self.class.to_s }
  json['id'] = @id ;
  json['type'] = @type ;
  json['pos'] = [@pos.x, @pos.y]

  return json ;
end
