def isCloseEnough(location, checkLaneP = false, margin = 1.0)
  if(location.edge == self.edge) then
    d = location.posOnLane - self.posOnLane ;
    if(Geo2D.abs(d) <= margin) then
      if(checkLaneP) then
        return location.laneIndex == self.laneIndex ;
      else
        return true ;
      end
    else
      return false ;
    end
  else
    return false ;
  end
end
