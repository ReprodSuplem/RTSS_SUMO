def toJson(obj)
  if(obj.is_a?(Array))
    return obj.map{|elm| toJson(elm)}
  elsif(obj.is_a?(Hash))
    newObj = {} ;
    obj.each{|key, value| newObj[key] = toJson(value)}
    return newObj ;
  else
    return (obj.respond_to?(:toJson) ?
              obj.toJson() :
              obj) ;
  end
end
