      def scanResponse(buffer)
        # get result code
        scanResultCode(buffer) ;
        # get the number of subscription
        n = DataType_Integer.unpack!(buffer) ;
        # scan subscription
        @_response = [] ;
        (0...n).each{|i|
          r = scanSubscriptionResponse(buffer) ;
          @_response.push(r) ;
        }
        #
        return @_response ;
      end
