      def scanResponse(buffer)
        # get result code
        scanResultCode(buffer) ;
        # check result code
        if(@_resultCode[:id] != @domId) then
          raise("Do not match sent and recv ID in setVariable :" + 
                @domId.inspect + @_resultCode.inspect) ;
        end
        if(@_resultCode[:code].name != :ok) then
          raise("something wrong at result code:" +
                @_resultCode.inspect) ;
        end
      end
