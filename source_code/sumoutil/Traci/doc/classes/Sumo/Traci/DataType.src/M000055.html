      def actualSize(value)
        case(@packTemplate)
        when(:composed)
          return actualSizeComposed(value) ;
        when(:composedList)
          return actualSizeComposedList(value) ;
        when(nil)
          raise("no pack template are given:" + 
                self.inspect) ;
        else
          return @size ;
        end
      end
