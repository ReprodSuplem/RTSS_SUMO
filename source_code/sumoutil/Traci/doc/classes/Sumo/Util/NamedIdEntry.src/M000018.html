      def initialize(name, cname, *rest)
        @name = name ;
        @nameInConstant = cname ;
        @id = Sumo::Traci::Constant[@nameInConstant] ;
        if(@id.nil?) then
          if(@nameInConstant.is_a?(Numeric))
            @id = @nameInConstant ;
          else
            puts "Warning!!! unknown @nameInConstant: #{@nameInConstant}."
            @id = nil ;
          end
        end
      end
