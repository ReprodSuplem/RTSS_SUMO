    def prepareServer()
      port = Sumo::Util::scanFreeTcpPort(12345) ;
      Sumo::Util::waitTcpPortIsFree(port) ;
      sumoCom = "sumo-gui -v -S" ;
#      sumoCom << " --step-length 100" ;
      sumoCom << " -c #{SampleConfFile} --remote-port #{port}" ;
      p [:run, sumoCom]
      system(sumoCom + "&") ;
      return port ;
    end
